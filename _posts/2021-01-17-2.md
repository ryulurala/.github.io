## C++

```c++
#include <string>
#include <vector>
#include <iostream>

using namespace std;

// board N x N, 5 <= N <= 30
// board 각 칸 0 이상 100 이하, 0은 빈 칸
// moves M, 1 <= M <= 1000

int solution(vector<vector<int>> board, vector<int> moves) {
    int answer = 0;
    int boardSize = board.size();
    int movesLenth = moves.size();
    vector<int> stk;    // 스택 자료구조로 이용

    // todo
    for(int i=0; i<movesLenth; i++){
        int curr = moves[i]-1;
        int item = 0;
        for(int j=0; j<boardSize; j++){
            if(board[j][curr] == 0) continue;
            item = board[j][curr];
            board[j][curr] = 0;
            break;
        }
        if(item != 0){
            // 인형을 집었을 때
            if(!stk.empty()){
                // 바구니에 인형이 있다면

                if(item == stk.back()){
                    // 같으면
                    answer += 2;       // 터뜨리기
                    stk.pop_back();     // 바구니에서 터뜨린 인형 빼기
                }
                else{
                    // 다르면
                    stk.push_back(item);
                }
            }
            else {
                // 바구니가 비어있다면
                stk.push_back(item);
            }
        }
    }

    return answer;
}
```
